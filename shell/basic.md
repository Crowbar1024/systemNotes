# basic

```bash
# 当前路径
./

# 复制
# 文件夹复制前要加-r，-r yh/


# 重启
reboot

# 改名
mv a b

# 删除
rm a

# 解压缩
tar -zxvf a

# 点命令
# 通常用于重新执行刚修改的初始或文件，使之立即生效，而不必注销并重新登录。
source
.

# 多行注释
:'
yhnb
yhnb
'
<<hhh
yhnb
yhnb
hhh

# 这个文件是每个用户登录时都会运行的环境变量设置，当用户第一次登录时,该文件被执行。
# 并从/etc/profile.d目录的配置文件中搜集shell的设置。
/etc/profile

# 全局检索关键字
grep 









```
